# vimfiles


    git clone https://github.com/emarkham/vimfiles.git ~/.vim  
    ln -s ~/.vim/vimrc ~/.vimrc  
    ln -s ~/.vim/gvimrc ~/.gvimrc  
    cd ~/.vim  
    git submodule init  
    git submodule update
